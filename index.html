int[] tri_R = new int [] {
  2,3,4,5,6,5,4,3
};
int[] tri_L = new int [] {
  3,4,5,6,5,4,3,2
};

int RED = 0;

color R0 = color (225,60,30);
color R1 = color (228,40,27);
color R2 = color (195,0,0);
color R3 = color (165,0,0);
color R4 = color (195,0,0);
color R5 = color (225,60,30);

void setup()
{
  background(255);
  size(600,380);
  frameRate(15);
  smooth();
}

void draw()
{
  noStroke();
  fill(0);


  for (int i = 0; i < 8; i++)
  {
    RED++;
    for (int j = 0; j < tri_R[i]; j++)
    {
      
      if(i<5)
      {
        
        switch ((RED+1) % 6)
        {
        case 0 :
          fill(R0); 
          break;
        case 1 :
          fill(R1); 
          break;
        case 2 :
          fill(R2); 
          break;
        case 3 :
          fill(R3); 
          break;
        case 4 :
          fill(R4); 
          break;
        case 5 :
          fill(R5); 
          break;
        }
        triangle(240 + i * 20, 165 + j * 30 - 15 *i, -20 + 240  + i * 20, -15 + 165 + j * 30 - 15 *i, -20 + 240 + i * 20, 15 + 165 + j * 30 - 15 *i);
      }

      else
      {
        
        switch ((RED+2) % 6)
        {
        case 0 :
          fill(R0); 
          break;
        case 1 :
          fill(R1); 
          break;
        case 2 :
          fill(R2); 
          break;
        case 3 :
          fill(R3); 
          break;
        case 4 :
          fill(R4); 
          break;
        case 5 :
          fill(R5); 
          break;
        }
        triangle(240 + i * 20, 120 + j * 30 + 15 * (i-5), -20 + 240  + i * 20, -15 + 120 + j * 30 + 15 * (i-5), -20 + 240 + i * 20, 15 + 120 + j * 30 + 15 * (i-5));
      }
    }
    for (int k = 0; k < tri_L[i]; k++)
    {
      
      if(i < 4)
      {
        
        switch ((RED+3) % 6)
        {
        case 0 :
          fill(R0); 
          break;
        case 1 :
          fill(R1); 
          break;
        case 2 :
          fill(R2); 
          break;
        case 3 :
          fill(R3); 
          break;
        case 4 :
          fill(R4); 
          break;
        case 5 :
          fill(R5); 
          break;
        }
        triangle(220 + i * 20, 150 + k * 30 - 15 *i, 20 + 220  + i * 20, -15 + 150 + k * 30 - 15 *i, 20 + 220 + i * 20, 15 + 150 + k * 30 - 15 *i);
      }
      else
      {
        
        switch ((RED+4) % 6)
        {
        case 0 :
          fill(R0); 
          break;
        case 1 :
          fill(R1); 
          break;
        case 2 :
          fill(R2); 
          break;
        case 3 :
          fill(R3); 
          break;
        case 4 :
          fill(R4); 
          break;
        case 5 :
          fill(R5); 
          break;
        }
        triangle(220 + i * 20, 120 + k * 30 + 15 * (i-4), 20 + 220  + i * 20, -15 + 120 + k * 30 + 15 * (i-4), 20 + 220 + i * 20, 15 + 120 + k * 30 + 15 * (i-4));
      }
    }
  }
  RED+=3;
  RED = RED % 6;
}
